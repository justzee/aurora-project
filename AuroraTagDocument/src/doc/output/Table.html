Tag Table

自适应宽度和高度的图表控件标签

列的属性可参阅Table[Column]

<a:table bindTarget="ds" rowRenderer="fun"
	width="300" percentWidth="90" title="tt">
	<a:columns>
		<a:column editor="tf" name="fieldName"
		renderer="fun" footerRenderer="fun"
		width="30" percentWidth="10" align="center"/>
	</a:columns>
	<a:editors>
		<a:textField id="tf"/>
	</a:editors>
</a:table>
